









\begin{restatable}[Optimal policy in the minimal $d$-reduction; {\citet{nilsson2000evaluating}}]{lemma}{lemmanonrequisiteremoval}%
\label{le:dred-performance}
Let 
$\calM$ be a model on 
$\calG$, a CID \withSR, 
that has $d$-reduction $\calG'$, 
Then, 
any optimal strategy for $\calM(\calG')$ is also optimal for $\calM$.
\end{restatable}
